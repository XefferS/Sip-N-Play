        <h1>Feladat átvétele</h1>
        <p>A következő feladatot a játékmestertől vehetitek át! A feladat átvétele után nyomjatok a „Tovább” gombra!</p>
        <button id="next-button" class="button">Tovább</button>
        <div id="error-message"></div>
    </main>

    <script>
        const nextButton = document.getElementById('next-button');
        const errorMessage = document.getElementById('error-message');
        const delayMs = 30 * 1000; // 30 másodperc

        const pageLoadTime = Date.now();

        nextButton.addEventListener('click', () => {
            const timeElapsed = Date.now() - pageLoadTime;

            if (timeElapsed < delayMs) {
                errorMessage.textContent = 'Nem túl korai ez?';
            } else {
                alert('A történet hamarosan folytatódik!');
                // window.location.href = 'kovetkezo_oldal.html';
            }
        });
    </script>
